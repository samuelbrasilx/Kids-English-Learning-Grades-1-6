let ttsVoices = [];

function ttsSupported(){ return typeof window.speechSynthesis !== "undefined" && typeof window.SpeechSynthesisUtterance !== "undefined"; }

 

function loadTTSPrefs(){

try{

const p = JSON.parse(localStorage.getItem(TTS_KEY) || "{}");

if(typeof p.enabled === "boolean") ttsEnable.checked = p.enabled;

if(typeof p.rate === "number") { ttsRate.value = String(p.rate); ttsRateVal.textContent = String(p.rate); }

if(typeof p.pitch === "number") { ttsPitch.value = String(p.pitch); ttsPitchVal.textContent = String(p.pitch); }

if(typeof p.voiceName === "string") ttsVoice.dataset.prefer = p.voiceName;

}catch{}

}

function saveTTSPrefs(){

const prefs = {

enabled: ttsEnable.checked,

voiceName: ttsVoice.value || "",

rate: parseFloat(ttsRate.value),

pitch: parseFloat(ttsPitch.value)

};

localStorage.setItem(TTS_KEY, JSON.stringify(prefs));

}

 

function populateVoices(){

if(!ttsSupported()) return;

ttsVoices = window.speechSynthesis.getVoices();

const english = ttsVoices.filter(v => /en[-_]/i.test(v.lang));

const list = english.length ? english : ttsVoices;

 

ttsVoice.innerHTML = "";

list.forEach(v=>{

const opt = document.createElement("option");

opt.value = v.name;

opt.textContent = `${v.name} (${v.lang})`;

ttsVoice.appendChild(opt);

});

 

const prefer = ttsVoice.dataset.prefer;

if(prefer && list.some(v=>v.name===prefer)){ ttsVoice.value = prefer; }

else if(list.length){ ttsVoice.value = list[0].name; }

}

 

function ttsSpeak(text){

if(!ttsSupported() || !ttsEnable.checked) return;

if(!text || !text.trim()) return;

window.speechSynthesis.cancel();

const u = new SpeechSynthesisUtterance(text);

u.rate = parseFloat(ttsRate.value);

u.pitch = parseFloat(ttsPitch.value);

const v = (ttsVoices.find(v => v.name === ttsVoice.value) || null);

if(v) u.voice = v;

window.speechSynthesis.speak(u);

}

 

function ttsDescribeQuizQuestion(){

try{

const q = (questionEl.textContent || "").replace(/^Q\d+:\s*/,'');

const opts = [...optionsEl.querySelectorAll("button")].map((b,i)=>`Option ${i+1}: ${b.textContent.replace(/^\d+\.\s*/,'')}`);

const txt = [q, ...opts].join(". ");

ttsSpeak(txt);

}catch{}

}

function ttsDescribeSight(){

const p = document.getElementById("swPrompt").textContent || "";

const choices = [...document.querySelectorAll("#swOptions button")].map(b=>b.textContent).join(", ");

if(p.trim()) ttsSpeak(`${p}. Choices: ${choices}.`);

}

function ttsDescribePhonics(){

const pat = document.getElementById("phTarget").textContent || "";

const choices = [...document.querySelectorAll("#phOptions button")].map(b=>b.textContent).join(", ");

ttsSpeak(`Find a word that starts with ${pat}. Choices: ${choices}.`);

}

function ttsDescribeSentence(){

ttsSpeak("Build the sentence by clicking the words in order.");

}

function ttsDescribeActive(){

const active = document.querySelector(".tab.active")?.dataset.tab;

if(active === "quiz") ttsDescribeQuizQuestion();

else if(active === "sight") ttsDescribeSight();

else if(active === "phonics") ttsDescribePhonics();

else if(active === "sentence") ttsDescribeSentence();

}

loadTTSPrefs();

populateVoices();

window.speechSynthesis.onvoiceschanged = populateVoices;

 

ttsEnable.addEventListener("change", saveTTSPrefs);

ttsPitch .addEventListener("input", ()=>{ ttsPitchVal.textContent = ttsPitch.value; saveTTSPrefs(); });

 

ttsReplay.addEventListener("click", ttsDescribeActive);

document.getElementById("ttsControls").style.display = "none";

/* ===== Start session ===== */

$id("startBtn").addEventListener("click", startSession);

function startSession(){

score=0;qIndex=0;missed=[];

scoreEl.textContent="0"; feedbackEl.textContent=""; remarkEl.textContent="";

missedWrap.style.display="none"; missedList.innerHTML="";

currentQuestions=[...(questionBank[currentLevel]||[])].sort(()=>Math.random()-0.5).slice(0,10);

renderQuestion();

ttsSpeak(`Starting Level ${currentLevel}. Ten questions.`);

}

 

/* ===== Render question & check ===== */

function renderQuestion(){

if(qIndex>=10||qIndex>=currentQuestions.length){

questionEl.textContent="🏆 Session Complete!";

optionsEl.innerHTML="";

feedbackEl.textContent=`Final Score: ${score} / 10`;

remarkEl.textContent = remarkFor(score);

if(missed.length){

missedWrap.style.display="block";

missedList.innerHTML="";

missed.forEach(t=>{const li=document.createElement("li"); li.textContent=t; missedList.appendChild(li);});

}

barFill.style.width="100%"; qNum.textContent="10";

updateProgress(currentLevel, score);

 

const status=grandTotal();

certHint.style.display = status.done ? "none" : "block";

ttsSpeak(`Session complete. Final score ${score} out of 10. ${remarkEl.textContent}`);

if(currentLevel===12 && status.done){ showCertificate(true); }

return;

}

const q=currentQuestions[qIndex];

qNum.textContent=String(qIndex+1); lvNow.textContent=String(currentLevel);

questionEl.textContent=`Q${qIndex+1}: ${q.q}`;

correctAnswer=q.a;

const opts=[...q.options].sort(()=>Math.random()-0.5);

optionsEl.innerHTML="";

opts.forEach((opt,i)=>{

const btn=document.createElement("button");

btn.textContent=`${i+1}. ${opt}`;

btn.addEventListener("click",()=>checkAnswer(opt,q));

optionsEl.appendChild(btn);

});

barFill.style.width=Math.round((qIndex/10)*100)+"%";

ttsDescribeQuizQuestion();

}

function checkAnswer(ans,q){

if(ans===correctAnswer){

score++; scoreEl.textContent=String(score);

feedbackEl.textContent="✅ Correct!";

ttsSpeak("Correct!");

}else{

feedbackEl.textContent="❌ Wrong!";

missed.push(`${q.q} Answer: ${q.a}`);

ttsSpeak(`Wrong. The correct answer is: ${q.a}.`);

}

qIndex++; setTimeout(renderQuestion,800);

}

 

/* ===== Remarks by score ===== */

function remarkFor(s){

if(s===10) return "Excellent, you rock! Perfect score!";

if(s>=8) return "Great work! Youre so close to perfectreview the tricky ones and go for 10.";

if(s>=6) return "Good job! Youre on the right trackone more round and youll level up.";

if(s>=4) return "Nice start! Youre warming upreview the ones in red and try again.";

return "Keep goingevery try makes you stronger! Lets practice a little more and youll see big progress.";

}

}

/* ========= SIGHT WORD SPRINT ========= */

const swDur=$id("swDur"), swGrade=$id("swGrade"), swStart=$id("swStart"),

swBar=$id("swBar"), swHUD=$id("swHUD"), swPrompt=$id("swPrompt"),

swOptions=$id("swOptions"), swScore=$id("swScore"), swAtt=$id("swAtt"),

swAcc=$id("swAcc"), swFeedback=$id("swFeedback");

let swTimer=null, swLeft=0, swS=0, swA=0, swAnswer="";

const sightBanks={

1:["the","and","to","in","it","is","you","that","of","a","I","was","for","on","are","as","with","his","they","be","at","one","have","this","from"],

2:["about","could","every","find","first","give","just","know","little","live","many","other","people","right","their","there","think","very","want","where"],

3:["although","between","brought","certain","enough","further","however","instead","perhaps","through","toward","whether","without","yourself","knowledge","language"]

};

function pick(arr,n){const a=[...arr].sort(()=>Math.random()-0.5); return a.slice(0,n);}

swStart.addEventListener("click",()=>{

swS=0; swA=0; updateSW();

swLeft=parseInt(swDur.value,10);

nextSW();

clearInterval(swTimer);

swTimer=setInterval(()=>{

swLeft--; if(swLeft<0){ endSW(); return;}

swHUD.textContent=`Time left: ${swLeft}s`;

swBar.style.width=Math.round((1 - swLeft/parseInt(swDur.value,10))*100)+"%";

},1000);

ttsSpeak(`Sight word sprint started for grade group ${swGrade.value}.`);

});

function updateSW(){ swScore.textContent=String(swS); swAtt.textContent=String(swA); swAcc.textContent=(swA?Math.round(100*swS/swA):0)+"%"; }

function nextSW(){

const bank = (swGrade.value==="1")?sightBanks[1]:(swGrade.value==="2")?sightBanks[2]:sightBanks[3];

const [target]=pick(bank,1);

swAnswer=target;

swPrompt.textContent=`Tap: ${target}`;

const distractors=pick(bank.filter(w=>w!==target),3);

const opts=[...distractors, target].sort(()=>Math.random()-0.5);

swOptions.innerHTML="";

opts.forEach((w)=>{

const b=document.createElement("button");

b.className="cta"; b.style.margin="6px"; b.textContent=w;

b.addEventListener("click",()=>submitSW(w));

swOptions.appendChild(b);

});

ttsDescribeSight();

}

function submitSW(word){

if(swLeft<=0) return;

swA++;

if(word===swAnswer){ swS++; swFeedback.textContent="✅ Nice!"; ttsSpeak("Nice!"); }

else{ swFeedback.textContent=`❌ It was ${swAnswer}.`; ttsSpeak(`Try again. It was ${swAnswer}.`); }

updateSW(); nextSW();

}

function endSW(){

clearInterval(swTimer);

swHUD.textContent=`Done! Score ${swS}/${swA} Accuracy ${swAcc.textContent}`;

swPrompt.textContent=""; swOptions.innerHTML="";

ttsSpeak(`Sight word sprint complete. Score ${swS} of ${swA}.`);

}

let phAnswer="sh";

const phonicsPatterns=[

{pat:"sh", words:["shop","ship","shell","shut","shape","shine"], distract:["hop","sip","bell","hut","cape","line"]},

{pat:"ch", words:["chip","chat","chop","chess","chill"], distract:["sip","hat","top","less","bill"]},

{pat:"th", words:["thin","then","thank","think","thumb"], distract:["fin","pen","bank","pink","drum"]},

{pat:"ph", words:["phone","photo","phrase","phonics"], distract:["stone","potato","raise","sonics"]},

{pat:"qu", words:["quick","queen","quiet","quiz"], distract:["brick","green","quite","quilt"]},

{pat:"oa", words:["boat","coat","road","toad"], distract:["bat","cot","rod","tad"]},

{pat:"ee", words:["see","tree","seed","need"], distract:["sea","try","said","nod"]},

{pat:"oo", words:["book","look","moon","noon"], distract:["bark","lock","moan","noun"]},

];

function newPhonics(){

const item=phonicsPatterns[Math.floor(Math.random()*phonicsPatterns.length)];

const target=item.pat; phAnswer=target; phTarget.textContent=target;

const correct = item.words[Math.floor(Math.random()*item.words.length)];

const wrongs = [...item.distract].sort(()=>Math.random()-0.5).slice(0,3);

const opts=[correct,...wrongs].sort(()=>Math.random()-0.5);

phOptions.innerHTML="";

opts.forEach(w=>{

const b=document.createElement("button");

b.className="cta"; b.style.margin="6px"; b.textContent=w;

b.addEventListener("click",()=>submitPhonics(w, target));

phOptions.appendChild(b);

});

ttsDescribePhonics();

}

function submitPhonics(word,pat){

phAtt.textContent=String(parseInt(phAtt.textContent,10)+1);

if(word.startsWith(pat)){ phOk.textContent=String(parseInt(phOk.textContent,10)+1); phMsg.textContent="✅ Correct!"; ttsSpeak("Correct!"); }

else{ phMsg.textContent=`❌ ${word} doesnt start with ${pat}.`; ttsSpeak(`Not quite. ${word} does not start with ${pat}.`); }

newPhonics();

}

phNew.addEventListener("click",()=>{ phMsg.textContent=""; newPhonics(); });

newPhonics();
